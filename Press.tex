\pgfkeys{
        /press/dimension/height/.code = {#1},
        /press/dimension/height/.initial = {1 cm},
        /press/dimension/height ratio/.code = {#1},
        /press/dimension/height ratio/.initial = {0.7},
        /press/dimension/width ratio/.code = {#1},
        /press/dimension/width ratio/.initial = {0.6},
        /press/dimension/width top/.code = {#1},
        /press/dimension/width top/.initial = {1 cm},
        /press/dimension/width bottom/.code = {#1},
        /press/dimension/width bottom/.initial = {2 cm},
        /press/.is choice,
        /press/basic/.code = {#1},
        /press/basic/.default = {},
        /press/roller/.code = {
            \pgfkeys{/press/basic/.append code = {
                \pgfpointscale{0.5}{\north}
                \pgf@ya = \pgf@y

                \pgfpointscale{0.2}{\southright}
                \pgf@xa = \pgf@x
                
                \pgfpathmoveto{\pgfpoint{0}{\pgf@ya}}
                \pgfpatharc{0}{360}{\pgf@xa}
                \pgfpathclose
                \pgfpatharc{180}{540}{\pgf@xa}
                \pgfpathclose
                
                \pgfusepath{stroke}
            }
        }},
        /press/piston/.code = {
            \pgfkeys{/press/basic/.append code = {
                \pgfpointscale{0.9}{\north}
                \pgf@ya = \pgf@y

                \pgfpointscale{0.5}{\north}
                \pgf@yb = \pgf@y

                \pgfpointscale{0.6}{\northright}
                \pgf@xa = \pgf@x
                
                \pgfpathmoveto{\pgfpoint{0}{\pgf@ya}}
                \pgfpathlineto{\pgfpoint{0}{\pgf@yb}}
                
                \pgfpathmoveto{\pgfpoint{-\pgf@xa}{\pgf@yb}}
                \pgfpathlineto{\pgfpoint{\pgf@xa}{\pgf@yb}}
                
                \pgfusepath{stroke}
            }
        }},
        /press/pelletizing/.code = {
            \pgfkeys{/press/basic/.append code = {
                \pgfpointscale{0.5}{\north}
                \pgf@ya = \pgf@y

                \pgfpointscale{0.4}{\northright}
                \pgf@xa = \pgf@x
                
                \pgfpathcircle{\pgfpoint{0}{\pgf@ya}}{\pgf@xa}
                \pgfusepath{stroke}
            }
        }},
        /press/.default = {\pgfkeys{/pump/basic}}  
}


\pgfdeclareshape{press}{
    \savedanchor{\centerpoint}{
        \pgfpointorigin
    }
    \savedanchor{\north}{
        \pgfpointscale{\pgfkeysvalueof{/press/dimension/height ratio}}{\pgfpoint{0}{\pgfkeysvalueof{/press/dimension/height}}}
    }
    \savedanchor{\northright}{
        \pgfpointscale{\pgfkeysvalueof{/press/dimension/height ratio}}{\pgfpoint{0}{\pgfkeysvalueof{/press/dimension/height}}}
        \pgf@ya = \pgf@y
        \pgfpointscale{0.5}{\pgfpoint{\pgfkeysvalueof{/press/dimension/width top}}{0}}
        \pgf@y = \pgf@ya
    }
    \savedanchor{\southright}{
        \pgfpointscale{\pgfkeysvalueof{/press/dimension/height ratio} - 1}{\pgfpoint{0}{\pgfkeysvalueof{/press/dimension/height}}}
        \pgf@ya = \pgf@y
        \pgfpointscale{0.5}{\pgfpoint{\pgfkeysvalueof{/press/dimension/width bottom}}{0}}
        \pgf@y = \pgf@ya
    }
    
    \anchor{center}{
        \centerpoint
    }
    \anchor{north}{
        \north
    }

    
    \backgroundpath{
        \pgfpathmoveto{\northright}
        \pgfpathlineto{\southright}
        \southright
        \pgfpathlineto{\pgfpoint{-\pgf@x}{\pgf@y}}
        \northright
        \pgfpathlineto{\pgfpoint{-\pgf@x}{\pgf@y}}
        \pgfpathclose

        \pgfusepath{stroke, clip}

        \southright
        
        \southright
        \pgfpathmoveto{\pgfpoint{-\pgf@x}{0}}
        \southright
        \pgfpathlineto{\pgfpoint{\pgf@x}{0}}

        \pgfpointscale{\pgfkeysvalueof{/press/dimension/width ratio}}{\southright}
        \pgf@xa = \pgf@x

        \southright
        \pgf@ya = \pgf@y

        \pgfpathmoveto{\pgfpoint{\pgf@xa}{\pgf@ya}}
        \pgfpathlineto{\pgfpoint{\pgf@xa}{0}}
        
        \pgfpathmoveto{\pgfpoint{-\pgf@xa}{\pgf@ya}}
        \pgfpathlineto{\pgfpoint{-\pgf@xa}{0}}

        \pgfusepath{stroke}
        
        \pgfkeys{/press/basic}
    
        \pgfusepath{stroke}
    }
}
